			focus_progress = {
				focus = SOV_decree_on_peace
				progress > 0.50
			}
		}
	}
	mean_time_to_happen = {
		days = 1
	}
	hidden = yes
	
	option = {
		name = ww1_russia.9.a

			transfer_state = 206
			transfer_state = 204
			transfer_state = 194
			transfer_state = 97
			transfer_state = 94
			transfer_state = 95
			transfer_state = 808
			add_state_claim = 207
			load_oob = "BLR_1918"
		}
		every_state = {
			limit = { is_core_of = BLR }
			remove_core_of = SOV
			remove_core_of = RUS
		}
		hidden_effect = {
			if = {
				limit = { GER = { has_war_with = RUS } }
				GER = {
					diplomatic_relation = {
						country = BLR
						relation = military_access
						active = yes
					}
				}
				RUS = {
					diplomatic_relation = {
						country = BLR
						relation = military_access
						active = yes
					}
				}
			}
			if = {
				limit = { AUH = { has_war_with = RUS } }
				AUH = {
					diplomatic_relation = {
						country = BLR
						relation = military_access
						active = yes
					}
				}
			}
		}
		GER = { country_event = { id = ww1_russia.11 days = 1 } }
		RUS = { country_event = { id = ww1_russia.11 days = 1 } }
		SOV = { country_event = { id = ww1_russia.11 days = 1 } }

	}
