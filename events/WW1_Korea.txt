# Korea's Declaration of Independence if Japan loses the Great War
country_event = {
	id = ww1_korea.1
	title = ww1_korea.1.t
	desc = ww1_korea.1.d
	fire_only_once = yes
	trigger = {
		tag=JAP
		JAP= { has_country_flag = great_war_loser }
		OR= {
			RUS= { has_country_flag = great_war_winner }
			CHI= { has_country_flag = great_war_winner }
			QIN= { has_country_flag = great_war_winner }
		}
		OR = {
			KOR = {
				is_subject_of = JAP
			}

			NOT = { country_exists = KOR }
		}

		hidden_effect = {
			release = KOR

			KOR = {
				set_country_flag = korean_independence

				set_politics = {
					ruling_party = democratic
					elections_allowed = no
				}
			}
		}
	}
	option = {
		name = ww1_korea.1.a
		ai_chance = {
			factor = 100
		}
	}
}

	is_triggered_only = yes
	option = {
		name = ww1_korea.19.a
		ai_chance = {
			factor = 80
		}

		every_country = {
			limit = {
				OR = {
					tag = SOV
					tag = RUS
				}
				OR = {
					country_exists = SOV
					country_exists = RUS
				}
			}

			add_state_core = 537
			add_state_core = 555
		}


		JAP = {
			every_owned_state = {
				limit = { is_core_of = ROOT is_owned_by = PREV }
				set_state_flag = korea_core_state_conference

					transfer_state = PREV
				}
				remove_state_core = PREV
			}
		}
		
		every_country = {
			limit = {
				is_subject_of = JAP
			}

				limit = {
					is_claimed_by = ROOT

				}
				ROOT = {
					transfer_state = PREV
				}
			}
		}

				is_subject_of = JAP
